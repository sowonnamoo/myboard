import { initializeApp } from "https://www.gstatic.com/firebasejs/9.22.0/firebase-app.js"; import { getFirestore, collection, addDoc } from "https://www.gstatic.com/firebasejs/9.22.0/firebase-firestore.js"; const firebaseConfig = { /* 설정 그대로 */ }; const app = initializeApp(firebaseConfig); const db = getFirestore(app); // 페이지 로드 시 URL 파라미터 받아서 채우기 window.onload = () => { const params = new URLSearchParams(window.location.search); const fullTitle = params.get('productName'); const imgCode = params.get('imgCode'); if (fullTitle) { const cleanData = fullTitle.replace(/[()]/g, ''); const parts = cleanData.split('님'); const dataStr = parts.length > 1 ? parts[1].trim() : parts[0].trim(); const items = dataStr.split('/'); if (items[0]) document.getElementById('productName').value = items[0].trim(); if (items[1]) document.getElementById('qty').value = items[1].trim(); if (items[2]) document.getElementById('size').value = items[2].trim(); if (items[3]) document.getElementById('price').value = items[3].trim(); if (parts[0]) document.getElementById('name').value = parts[0].trim(); } if (imgCode) document.getElementById('imgCode').value = imgCode; }; // [핵심] 'jae' 컬렉션으로 저장하는 로직 window.submitOrder = async (type) => { try { await addDoc(collection(db, "jae"), { // 여기서 'jae' 컬렉션 지정 productName: document.getElementById('productName').value, quantity: document.getElementById('qty').value, size: document.getElementById('size').value, price: document.getElementById('price').value, imgCode: document.getElementById('imgCode').value, type: type, name: document.getElementById('name').value, phone: document.getElementById('phone').value.replace(/[^0-9]/g, ''), address: document.getElementById('address').value + " " + document.getElementById('addressDetail').value, timestamp: new Date().toLocaleString() }); alert('주문 접수가 완료되었습니다 (jae 저장).'); } catch (e) { alert('오류: ' + e.message); } };
