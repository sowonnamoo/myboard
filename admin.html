window.maskInfo = async function() { if (!confirm("마스킹 작업을 시작합니다. 콘솔창(F12)을 확인하세요!")) return; const status = document.getElementById("status"); const thirtyDaysAgo = new Date(); thirtyDaysAgo.setDate(thirtyDaysAgo.getDate() - 30); console.log("기준 날짜 (이 날짜보다 이전이면 마스킹):", thirtyDaysAgo); const q = query(collection(db, "boards")); const snapshot = await getDocs(q); let count = 0; snapshot.forEach((docSnap) => { const data = docSnap.data(); // 데이터 확인용 로그 console.log("---"); console.log("글 번호:", docSnap.id); console.log("글 날짜(createdAt):", data.createdAt?.toDate()); const createdAt = data.createdAt?.toDate(); if (createdAt && createdAt < thirtyDaysAgo) { console.log("✅ 조건 만족: 마스킹 대상입니다."); // 여기서 수정할 데이터 로직이 실행됨 // (실제 마스킹 실행하려면 여기에 batch.update 로직을 넣으세요) count++; } else { console.log("❌ 조건 불만족: 마스킹 대상이 아닙니다."); } }); status.innerText = `확인 완료! 마스킹 대상 게시물: ${count}개 (상세 내용은 F12 콘솔창 확인)`; };
